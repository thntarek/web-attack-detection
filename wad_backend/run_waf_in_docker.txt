from fastapi import FastAPI, Request

app = FastAPI()

@app.get("/submit")
async def analyze(data: str, request: Request):
    # This captures the 'data' query parameter (?data=...) smoothly
    print(f"Received clean data from proxy: {data}")

    # You can also capture any custom headers passed down by the WAF
    modsec_id = request.headers.get("X-ModSec-ID")

    return {
        "status": "Success",
        "message": "FastAPI processed this request cleanly!",
        "received_data": data
    }
